export default function HostListEditorPage({ params }: { params: { slug: string } }) {
  const router = useRouter();
  const searchParams = useSearchParams();
  const token = searchParams.get("token");
  
  const [list, setList] = useState<GiftList | null>(null);
  const [activeTab, setActiveTab] = useState<"itens" | "recados">("itens");
  const [searchTerm, setSearchTerm] = useState("");
  const [isEditingMetadata, setIsEditingMetadata] = useState(false);
  const [isScannerOpen, setIsScannerOpen] = useState(false);
  const [scanStatus, setScanStatus] = useState<"idle" | "success" | "error">("idle");
  const inputRef = useRef<HTMLInputElement>(null);

  // Initial load
  useEffect(() => {
    const foundList = getGiftListBySlug(params.slug);
    if (foundList) {
      setList(foundList);
    }
  }, [params.slug]);

  // Keep focus on scanner input for physical scanner support
  useEffect(() => {
    if (!isEditingMetadata && !isScannerOpen) {
      inputRef.current?.focus();
    }
  }, [isEditingMetadata, isScannerOpen]);

  // Scanner Logic (Auto-match EAN)
  useEffect(() => {
    if (searchTerm.length >= 8) {
      const product = allProducts.find(p => p.ean === searchTerm);
      if (product) {
        handleAddProduct(product);
        setSearchTerm("");
        setScanStatus("success");
        setTimeout(() => setScanStatus("idle"), 2000);
      } else if (searchTerm.length >= 13) {
        setScanStatus("error");
        setTimeout(() => setScanStatus("idle"), 2000);
        setSearchTerm("");
      }
    }
  }, [searchTerm]);

  const handleAddProduct = (product: Product) => {
    if (!list) return;
    
    // Check if already in list
    if (list.items.some((i: GiftListItem) => i.productId === product.id)) {
      alert("Este produto já está na sua lista!");
      return;
    }

    const newItem: GiftListItem = {
      id: `item-${Date.now()}`,
      productId: product.id,
      note: "Item escolhido na loja via scanner.",
      status: "disponivel"
    };

    const updated = { ...list, items: [newItem, ...list.items] };
    setList(updated);
  };

  const handleMockScan = () => {
    // Pick a random product from allProducts to simulate a successful scan
    const availableProducts = allProducts.filter(p => !list?.items.some((i: GiftListItem) => i.productId === p.id));
    if (availableProducts.length === 0) {
      alert("Você já adicionou todos os produtos disponíveis!");
      setIsScannerOpen(false);
      return;
    }
    
    const randomProduct = availableProducts[Math.floor(Math.random() * availableProducts.length)];
    handleAddProduct(randomProduct);
    
    setScanStatus("success");
    setTimeout(() => {
      setScanStatus("idle");
      setIsScannerOpen(false);
    }, 1500);
  };

  const handleRemoveItem = (itemId: string) => {
    if (!list) return;
    setList({
      ...list,
      items: list.items.filter((i: GiftListItem) => i.id !== itemId)
    });
  };

  if (!list) {
    return (
      <div className="flex h-screen flex-col items-center justify-center p-8 text-center bg-[#fefcfb]">
        <div className="h-20 w-20 rounded-full bg-[#f7f2ed] flex items-center justify-center text-[#8c6d45] mb-6 shadow-xl shadow-[#b08d57]/10">

          <img 
            src={list.photo} 
            className="h-full w-full object-cover grayscale-[20%] opacity-80" 
            alt={list.brideName} 
          />
          <div className="absolute inset-0 bg-gradient-to-t from-[#2a2421] via-transparent to-transparent flex flex-col justify-end p-6 md:p-12">

                {new Date(list.eventDate).toLocaleDateString('pt-BR')}

              <h2 className="font-serif text-3xl md:text-6xl text-white tracking-tight leading-none">{list.brideName}</h2>

"use client";

import Link from "next/link";
import { useSearchParams } from "next/navigation";
import { useEffect, useState } from "react";
import { Heart, Gift, Calendar, MapPin, Share2, Info, Check, X, ArrowLeft } from "lucide-react";
import { ListaItem } from "@/components/listas/ListaItem";
import { getGiftListBySlug, getProductById, getTenantBySlug } from "@/lib/mock-data";
import { GiftList, GiftListItem, Product } from "@/types";
import { formatDate, formatCurrency } from "@/lib/utils";

export default function GuestListPage({ params }: { params: { slug: string } }) {
  const searchParams = useSearchParams();
  const [list, setList] = useState<GiftList | null>(null);
  const [selectedItem, setSelectedItem] = useState<{ item: GiftListItem, product: Product } | null>(null);
  const [guestInfo, setGuestInfo] = useState({ name: "", message: "" });
  const [isSuccess, setIsSuccess] = useState(false);
  const tenant = getTenantBySlug("le-blanc");
  const isHostPreview = searchParams.get("preview") === "host";
  const returnToken = searchParams.get("returnToken");

  useEffect(() => {
    const foundList = getGiftListBySlug(params.slug);
    if (foundList) {
      setList(foundList);
    }
  }, [params.slug]);

  if (!list) {
    return (
      <div className="flex h-screen items-center justify-center p-12 text-center bg-[#fefcfb]">
        <div className="space-y-4">
          <Gift className="h-16 w-16 mx-auto text-[#b08d57] opacity-20" />
          <h2 className="font-serif text-3xl text-[#2a2421]">Lista não encontrada</h2>
          <p className="text-sm text-[#8c6d45] uppercase tracking-widest">Verifique o link ou entre em contato com os anfitriões.</p>
        </div>
      </div>
    );
  }

  const handleConfirmGift = () => {
    if (!guestInfo.name) {
      alert("Por favor, informe seu nome.");
      return;
    }
    
    // Simulate updating list status locally
    if (list && selectedItem) {
      const updatedItems = list.items.map((i: GiftListItem) => 
        i.id === selectedItem.item.id ? { ...i, status: "reservado" as const } : i
      );
      setList({ ...list, items: updatedItems });
      setIsSuccess(true);
    }
  };

  const closeOverlay = () => {
    setSelectedItem(null);
    setIsSuccess(false);
    setGuestInfo({ name: "", message: "" });
  };

  const handleShare = async () => {
    const shareUrl = window.location.href;
    const shareData = {
      title: `${tenant?.logoLabel ?? "Le Blanc"} | Lista de Presentes`,
      text: `Confira a lista de presentes de ${list.brideName}.`,
      url: shareUrl
    };

    try {
      if (navigator.share) {
        await navigator.share(shareData);
        return;
      }
    } catch (error) {
      if (error instanceof DOMException && error.name === "AbortError") {
        return;
      }
    }

    try {
      await navigator.clipboard.writeText(shareUrl);
      alert("Link copiado para compartilhar a lista.");
    } catch {
      window.prompt("Copie o link da lista:", shareUrl);
    }
  };

  const itemsWithProducts = list.items
    .map((item: GiftListItem) => {
      const product = getProductById(item.productId);
      return product ? { ...item, product } : null;
    })
    .filter(Boolean) as ({ product: Product } & GiftListItem)[];

  return (
    <div className="min-h-screen bg-[#fdfbf7] selection:bg-[#b08d57]/20 relative pb-40">
      <header className="sticky top-0 z-40 flex h-20 items-center justify-between border-b-2 border-[#b08d57]/20 bg-white/95 px-4 shadow-sm backdrop-blur-md md:px-8">
        <div className="flex w-24 md:w-32">
          {isHostPreview ? (
            returnToken ? (
              <Link
                href={`/lista/${params.slug}/editar?token=${encodeURIComponent(returnToken)}`}
                className="flex items-center gap-2 text-[#8c6d45] hover:text-[#5c4a33] transition-colors"
              >
                <ArrowLeft className="h-5 w-5" />
                <span className="hidden md:inline text-[10px] font-black uppercase tracking-widest">
                  Voltar
                </span>
              </Link>
            ) : (
              <button
                type="button"
                onClick={() => window.history.back()}
                className="flex items-center gap-2 text-[#8c6d45] hover:text-[#5c4a33] transition-colors"
              >
                <ArrowLeft className="h-5 w-5" />
                <span className="hidden md:inline text-[10px] font-black uppercase tracking-widest">
                  Voltar
                </span>
              </button>
            )
          ) : null}
        </div>

        <div className="flex flex-col items-center text-center">
          <h1 className="font-serif text-2xl md:text-3xl text-[#2a2421] tracking-[0.2em] uppercase leading-none">
            {tenant?.logoLabel ?? "Le Blanc"}
          </h1>
          <p className="mt-2 text-[8px] md:text-[9px] uppercase tracking-[0.5em] text-[#8c6d45] font-black">
            Lista do Convidado
          </p>
        </div>

        <div className="w-24 md:w-32" />
      </header>

      {/* Luxury Hero Invitation Section */}
      <section className="relative min-h-[80vh] flex items-center justify-center overflow-hidden">
        {/* Background Image Accent */}
        <div className="absolute inset-0 z-0">
          <img 
            src={list.photo} 
            className="w-full h-full object-cover grayscale-[30%] opacity-40 scale-110 blur-sm" 
            alt="Background" 
          />
          <div className="absolute inset-0 bg-gradient-to-b from-[#fdfbf7]/40 via-transparent to-[#fdfbf7]" />
        </div>

        <div className="relative z-10 max-w-5xl mx-auto px-6 text-center space-y-12 animate-in fade-in zoom-in duration-1000">
          <div className="space-y-4">
            <div className="flex items-center justify-center gap-4 text-[#8c6d45]">
              <div className="h-[1px] w-8 bg-[#b08d57]/30" />
              <span className="text-[10px] uppercase tracking-[0.5em] font-black">Uma Celebração Le Blanc</span>
              <div className="h-[1px] w-8 bg-[#b08d57]/30" />
            </div>
            <h1 className="font-serif text-6xl md:text-9xl text-[#2a2421] leading-tight tracking-tighter">
              {list.brideName.replace(" & ", " \n & ")}
            </h1>
          </div>

          <div className="max-w-xl mx-auto py-10 border-y border-[#b08d57]/10 relative">
            <Heart className="absolute -top-4 left-1/2 -translate-x-1/2 h-8 w-8 text-[#b08d57]/20 fill-current" />
            <p className="font-serif text-2xl md:text-3xl text-[#5c4a33] italic leading-relaxed">
              "{list.headline}"
            </p>
          </div>

          <div className="flex flex-col md:flex-row items-center justify-center gap-12 text-[#2a2421]">
            <div className="flex flex-col items-center gap-2">
              <span className="text-[9px] uppercase tracking-[0.3em] font-black text-[#8c6d45]/60 mb-1">Data do Evento</span>
              <div className="flex items-center gap-3 font-serif text-2xl md:text-3xl uppercase tracking-tighter">
                <Calendar className="h-5 w-5 text-[#b08d57]" />
                {formatDate(list.eventDate)}
              </div>
            </div>
            <div className="hidden md:block h-12 w-[1px] bg-[#b08d57]/20" />
            <div className="flex flex-col items-center gap-2">
              <span className="text-[9px] uppercase tracking-[0.3em] font-black text-[#8c6d45]/60 mb-1">Cidade / Local</span>
              <div className="flex items-center gap-3 font-serif text-2xl md:text-3xl uppercase tracking-tighter">
                <MapPin className="h-5 w-5 text-[#b08d57]" />
                {list.city}
              </div>
            </div>
          </div>
        </div>
      </section>

      {/* Gift Grid Section */}
      <section className="max-w-7xl mx-auto px-6 pt-24 space-y-20">
        <div className="flex flex-col md:flex-row items-end justify-between gap-8 border-b border-[#b08d57]/10 pb-12">
          <div className="space-y-2">
            <h2 className="font-serif text-4xl md:text-6xl text-[#2a2421]">Sugestões de Presentes</h2>
            <p className="text-[10px] md:text-xs uppercase tracking-[0.4em] text-[#8c6d45] font-black">
              Escolha um item especial para celebrar este momento
            </p>
          </div>
          <div className="flex items-center gap-6">
             <div className="flex items-center gap-2 text-[10px] font-bold uppercase tracking-widest text-[#a69b8f]">
               <Info className="h-4 w-4" />
               <span>Itens marcados já foram presenteados</span>
             </div>
          </div>
        </div>

        <div className="grid gap-x-10 gap-y-20 sm:grid-cols-2 lg:grid-cols-3">
          {itemsWithProducts.map((item) => (
            <ListaItem
              key={item.id}
              product={item.product}
              status={item.status}
              note={item.note || "Um presente escolhido com carinho para o nosso lar."}
              publicView
              onChoose={() => setSelectedItem({ item, product: item.product })}
            />
          ))}
        </div>
      </section>

      {/* Fixed Sticky Call to Action (Floating Guest Controls) */}
      <div className="fixed bottom-10 left-1/2 -translate-x-1/2 z-40 animate-in slide-in-from-bottom-2 duration-700">
        <div className="bg-white/80 backdrop-blur-3xl border-2 border-[#b08d57]/20 rounded-full py-4 px-8 shadow-2xl flex items-center gap-8 ring-8 ring-white/50">
          <div className="flex -space-x-4">
             {[1,2,3,4].map(i => (
               <div key={i} className="h-10 w-10 rounded-full border-2 border-white overflow-hidden bg-[#f7f2ed] shadow-inner">
                 <img src={`https://unavatar.io/instagram/wedding${i}`} className="h-full w-full object-cover grayscale" />
               </div>
             ))}
          </div>
          <p className="text-[10px] font-black uppercase tracking-widest text-[#5c4a33]">
             <span className="text-[#8c6d45]">{list.items.filter((i: GiftListItem) => i.status === 'comprado').length} amigos</span> já presentearam!
          </p>
          <div className="h-8 w-[1px] bg-[#b08d57]/20" />
          <button
            onClick={handleShare}
            className="flex items-center gap-2 text-[10px] font-black uppercase tracking-[0.2em] text-[#8c6d45] hover:text-[#2a2421] transition-colors"
          >
            <Share2 className="h-4 w-4" />
            Compartilhar
          </button>
        </div>
      </div>

      {/* Gifting Modal Overlay */}
      {selectedItem && (
        <div className="fixed inset-0 z-50 flex items-center justify-center p-6 bg-[#2a2421]/70 backdrop-blur-md animate-in fade-in duration-500">
          <div className="w-full max-w-2xl bg-white rounded-[4rem] p-10 md:p-16 shadow-2xl relative animate-in slide-in-from-bottom-12 duration-700 overflow-hidden">
            
            <button onClick={closeOverlay} className="absolute top-10 right-10 text-[#a69b8f] hover:text-[#2a2421] transition-colors">
              <X className="h-8 w-8" />
            </button>

            {!isSuccess ? (
              <div className="grid md:grid-cols-2 gap-12 items-center">
                <div className="space-y-8">
                  <div className="aspect-[4/5] rounded-[2.5rem] overflow-hidden bg-[#f7f2ed] shadow-lg border border-[#b08d57]/10">
                    <img src={selectedItem.product.images[selectedItem.product.mainImageIndex]} className="h-full w-full object-cover" />
                  </div>
                  <div className="text-center space-y-2">
                    <h4 className="font-serif text-2xl text-[#2a2421]">{selectedItem.product.name}</h4>
                    <p className="text-lg text-[#8c6d45] font-semibold">{formatCurrency(selectedItem.product.price)}</p>
                  </div>
                </div>

                <div className="space-y-8">
                  <div className="space-y-2">
                    <h3 className="font-serif text-3xl text-[#2a2421]">Reservar Presente</h3>
                    <p className="text-[10px] uppercase tracking-[0.3em] font-black text-[#8c6d45]">Compromisso de compra na loja física</p>
                  </div>

                  <div className="space-y-6">
                    <div className="bg-[#f7f2ed] border-l-4 border-[#8c6d45] p-5 rounded-r-2xl">
                       <p className="text-[11px] text-[#5c4a33] leading-relaxed italic">
                         "Este item ficará reservado em seu nome. Visite a **Le Blanc** para concluir a compra e entregar seu presente com carinho."
                       </p>
                    </div>

                    <div className="space-y-3">
                      <label className="text-[10px] uppercase tracking-widest font-black text-[#8c6d45] ml-1">Seu Nome</label>
                      <input 
                        className="w-full rounded-2xl border-2 border-[#b08d57]/10 bg-[#fdfbf7] p-5 text-sm font-medium focus:border-[#8c6d45] outline-none transition-all"
                        placeholder="Ex: Família Sampaio"
                        value={guestInfo.name}
                        onChange={(e) => setGuestInfo({ ...guestInfo, name: e.target.value })}
                      />
                    </div>
                    <div className="space-y-3">
                      <label className="text-[10px] uppercase tracking-widest font-black text-[#8c6d45] ml-1">Mensagem para o Casal</label>
                      <textarea 
                        className="w-full rounded-2xl border-2 border-[#b08d57]/10 bg-[#fdfbf7] p-5 text-sm font-medium focus:border-[#8c6d45] outline-none transition-all"
                        placeholder="Escreva seus votos de felicidade..."
                        rows={3}
                        value={guestInfo.message}
                        onChange={(e) => setGuestInfo({ ...guestInfo, message: e.target.value })}
                      />
                    </div>
                    <button 
                      onClick={handleConfirmGift}
                      className="w-full py-6 bg-[#2a2421] text-white rounded-[2rem] text-[10px] font-black uppercase tracking-[0.3em] hover:bg-black transition-all shadow-xl shadow-black/10 flex items-center justify-center gap-3"
                    >
                      Confirmar Reserva
                      <Check className="h-4 w-4" />
                    </button>
                  </div>
                </div>
              </div>
            ) : (
              <div className="text-center space-y-10 py-10 animate-in zoom-in duration-500">
                <div className="h-24 w-24 bg-[#8c6d45]/10 rounded-full flex items-center justify-center mx-auto text-[#8c6d45] shadow-xl shadow-[#8c6d45]/10">
                  <Heart className="h-12 w-12 stroke-[2.5px] fill-[#8c6d45]" />
                </div>
                <div className="space-y-4">
                  <h3 className="font-serif text-5xl text-[#2a2421]">Reserva Confirmada!</h3>
                  <p className="text-[#8c6d45] font-serif text-xl italic max-w-sm mx-auto leading-relaxed px-4">
                    "Obrigado pelo carinho. Sua reserva foi registrada e em breve nos veremos na Le Blanc para finalizar esse momento especial!"
                  </p>
                </div>
                <div className="pt-8">
                  <button 
                    onClick={closeOverlay}
                    className="px-12 py-5 bg-[#2a2421] text-white rounded-full text-[10px] font-black uppercase tracking-widest hover:bg-black transition-all shadow-lg"
                  >
                    Voltar para a Lista
                  </button>
                </div>
              </div>
            )}

            <div className="absolute -bottom-10 left-1/2 -translate-x-1/2 w-48 h-1 bg-[#b08d57]/20 rounded-full" />
          </div>
        </div>
      )}
    </div>
  );
}
